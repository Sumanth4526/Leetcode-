class Solution {
    static int count=0;
    public int[] smallerNumbersThanCurrent(int[] nums) {
        
        int[] res=new int[nums.length];
        int element=0;
        for(int i=0;i<nums.length;i++)
		{
			res[i]=nums[i];
			count=0;
            
			for(int j=0;j<nums.length;j++)
			{
				if(res[i]>nums[j])
				{
					count++;	
				}
        
			}	
            res[i]=count;
		}
        return res;
    }
}
